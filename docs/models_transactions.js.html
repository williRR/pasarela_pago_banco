import sql from 'mssql';
import { dbConfigPasarela } from '../config.js'; // Usa la configuracin que acabas de crear

/**
 * Llama al Stored Procedure 'sp_registrarTransaccion' para guardar el registro inicial
 * de una transacción en la base de datos con estado PENDIENTE.
 *
 * @async
 * @function registerInitialTransaction
 * @param {object} params - Parámetros de la transacción a registrar.
 * @param {string} params.merchantId - Identificador único del comercio. Máximo 50 caracteres.
 * @param {number} params.amount - Monto de la transacción con dos decimales (Decimal 18, 2).
 * @param {string} params.cardNumber - Número completo de la tarjeta de crédito para extraer los últimos 4 dígitos.
 * @param {string} [params.moneda='USD'] - Código ISO de la moneda de la transacción. Máximo 3 caracteres.
 * @returns {Promise<number>} Retorna una Promesa que se resuelve con el `int_transaccionid` generado por la base de datos.
 * @throws {Error} Lanza un error si falla la conexión o la ejecución del Stored Procedure.
 */
export async function registerInitialTransaction({ merchantId, amount, cardNumber, moneda = 'USD' }) {
    try {
        const pool = await sql.connect(dbConfigPasarela);
        const request = pool.request();
        
        // Asignacin de inputs para el SP
        request.input('merchantid', sql.VarChar(50), merchantId);
        request.input('monto', sql.Decimal(18, 2), amount);
        request.input('moneda', sql.VarChar(3), moneda);
        request.input('ultimos4', sql.VarChar(4), cardNumber.slice(-4));
        request.output('transaccionid', sql.Int);

        await request.execute('sp_registrarTransaccion');

        return request.output.transaccionid;

    } catch (err) {
        console.error("Error al registrar transaccin inicial:", err);
        throw new Error("Fallo al guardar la transaccin en la base de datos.");
    }
}

/**
 * Llama al Stored Procedure 'sp_actualizarEstadoTransaccion' para registrar el resultado final del banco.
 *
 * @async
 * @function updateTransactionStatus
 * @param {object} params - Parámetros de actualización.
 * @param {number} params.transactionId - El ID de la transacción generado previamente (int_transaccionid).
 * @param {string} params.status - El nuevo estado de la transacción (ej. APROBADO, RECHAZADO, ERROR). Se convierte a mayúsculas.
 * @param {string} params.message - Mensaje descriptivo del resultado de la transacción.
 * @param {number | null} [params.cuentaReferencia=null] - Número de referencia de la cuenta o del banco asociado a la transacción, si aplica.
 * @returns {Promise<void>} No retorna ningún valor, solo confirma la ejecución del SP.
 */
export async function updateTransactionStatus({ transactionId, status, message, cuentaReferencia = null }) {
    try {
        const pool = await sql.connect(dbConfigPasarela);
        const request = pool.request();
        
        // Asignacin de inputs para el SP
        request.input('transaccionid', sql.Int, transactionId);
        request.input('nuevoEstado', sql.VarChar(20), status.toUpperCase()); 
        request.input('mensaje', sql.VarChar(200), message);
        request.input('cuentareferencia', sql.Int, cuentaReferencia);

        await request.execute('sp_actualizarEstadoTransaccion');

    } catch (err) {
        console.error(`Error al actualizar estado de TX ${transactionId}:`, err);
        // La transaccin del banco ya se complet, no interrumpimos la respuesta final.
    }
}
